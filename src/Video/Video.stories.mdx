  "description": "A reusable component to display a video",
  "allOf": [
    {
      "$ref": "http://bigcontent.io/cms/schema/v1/core#/definitions/content"
    }
  ],

  "properties": {
    "video": {
      "type": "object",
      "title": "Video",
