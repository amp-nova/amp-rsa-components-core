```json
{ 
  "$schema": "http://json-schema.org/draft-07/schema#",
  "$id": "https://amp-rsa.amplience.com/cardList.json",
  "title": "Card List",
  "description": "A list of cards",
  "allOf": [
    {
      "$ref": "http://bigcontent.io/cms/schema/v1/core#/definitions/content"
    }
  ],
  "type": "object",
  "properties": {
    "header": {
      "type": "string",
      "title": "Header",
      "description": "Header/Title for the card list",
      "validation": "Please enter less than 50 characters",
      "minLength": 0,
      "maxLength": 50
    },
    "cards": {
      "type": "array",
      "title": "Cards",
      "description": "Please select the cards to display in this list",
      "items": {
        "allOf": [
          {
            "$ref": "http://bigcontent.io/cms/schema/v1/core#/definitions/content-link"
          },
          {
            "properties": {
              "contentType": {
                "title": "",
                "enum": [
                  "https://amp-rsa.amplience.com/card.json"
                ]
              }
            }
          }
        ]
      },
      "minItems": 1,
      "maxItems": 6
    }
  },
  "propertyOrder": ["header", "heroList", "slider", "cards", "style"]
}
```
